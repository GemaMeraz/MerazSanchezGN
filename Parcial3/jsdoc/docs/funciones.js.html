/**
 * Esta funcion recibe un numero que se va a elevar y devuelve el resultado
 * @param {number} numero Numero que se va a elevar
 * @param {number} exponente Potencia a la que se va a elevar el numero
 * 
 */
 export function potencia(numero,exponente){

}
